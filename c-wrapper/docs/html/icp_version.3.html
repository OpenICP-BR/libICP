NAME

       icp_version - Returns the libICP version as a NULL-terminated string

SYNOPSIS

         #include "libICP.h"

         char* icp_version();

EXAMPLE

               #include "libICP.h"

               int main() {
                   printf("%s\n", icp_version());
               }
